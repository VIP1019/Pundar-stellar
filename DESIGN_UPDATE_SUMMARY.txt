🚀 HOW TO USE

IMPORT UTILITIES:
┌────────────────────────────────────────────────────────────────────────
import com.example.pundarapp.ui.utils.Animations
import com.example.pundarapp.ui.utils.DesignTokens
└────────────────────────────────────────────────────────────────────────

USE DESIGN TOKENS:
┌────────────────────────────────────────────────────────────────────────
Column(modifier = Modifier.padding(Spacing.lg)) {
    Text("Title")
}
Surface(shape = RoundedCornerShape(Radius.md)) { }
Box(modifier = Modifier.shadow(Shadows.lg, shape = RoundedCornerShape(Radius.lg)))
└────────────────────────────────────────────────────────────────────────

USE ANIMATIONS:
┌────────────────────────────────────────────────────────────────────────
val scale by animateFloatAsState(
    targetValue = 1f,
    animationSpec = FastAnimation,
    label = "myAnimation"
)
Box(modifier = Modifier.graphicsLayer(scaleX = scale, scaleY = scale))
└────────────────────────────────────────────────────────────────────────
